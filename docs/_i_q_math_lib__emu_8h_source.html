456static inline int32_t __IQxmpy(int32_t a, int32_t b, int16_t shift) {
457 int64_t shiftOut;
458
459
460 shift = 32 - shift;
461
462 if (shift > 0) {
463 shiftOut = (((int64_t)a * (int64_t)b) >> shift);
464 } else if (shift < 0) {
465 shiftOut = (((int64_t)a * (int64_t)b) << -shift);
466 } else {
467 shiftOut = (((int64_t)a * (int64_t)b));
468 }
469
470 if (shiftOut > (int64_t)INT32_MAX) {
471 shiftOut = (int64_t)INT32_MAX;
472 }
473
474 if (shiftOut < (int64_t)INT32_MIN) {
475 shiftOut = (int64_t)INT32_MIN;
476 }
477
478
479 return (int32_t)shiftOut;
480}
